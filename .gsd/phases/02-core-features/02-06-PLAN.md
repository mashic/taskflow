    ```typescript
    @Component({
      selector: 'app-kanban-page',
      standalone: true,
      imports: [CdkDropListGroup, KanbanList, TaskCreateDialog],
      template: `
        <div class="kanban-header">
          <h1>{{ boardStore.selectedBoard()?.title }}</h1>
        </div>
        
        @if (listStore.isLoading() || taskStore.isLoading()) {
          <p>Loading board...</p>
        } @else {
          <div class="kanban-board" cdkDropListGroup>
            @for (list of listStore.sortedLists(); track list.id) {
              <app-kanban-list
                [list]="list"
                [tasks]="getTasksForList(list.id)"
                (onDrop)="handleDrop($event, list.id)"
                (onAddTask)="openTaskDialog(list.id)"
              />
            }
          </div>
        }
        
        @if (showTaskDialog()) {
          <app-task-create-dialog
            [listId]="selectedListId()!"
            [boardId]="boardId()!"
            (close)="closeTaskDialog()"
          />
        }
      `,
      styles: [`
        .kanban-board {
          display: flex;
          gap: 16px;
          padding: 16px;
          overflow-x: auto;
          align-items: flex-start;
        }
      `]
    })
    export class KanbanPage implements OnInit {
      boardStore = inject(BoardStore);
      listStore = inject(ListStore);
      taskStore = inject(TaskStore);
      private route = inject(ActivatedRoute);
      
      boardId = signal<string | null>(null);
      showTaskDialog = signal(false);
      selectedListId = signal<string | null>(null);
      
      ngOnInit() {
        const id = this.route.snapshot.paramMap.get('id')!;
        this.boardId.set(id);
        this.boardStore.setSelectedBoard(id);
        this.listStore.loadListsForBoard(id);
        this.taskStore.loadTasksForBoard(id);
      }
      
      getTasksForList(listId: string): Task[] {
        return this.taskStore.tasksByList().get(listId) || [];
      }
      
      handleDrop(event: CdkDragDrop<Task[]>, targetListId: string) {
        const task = event.item.data as Task;
        const newPosition = this.calculatePosition(event.currentIndex, targetListId);
        
        if (event.previousContainer === event.container) {
          // Reorder within same list
          this.taskStore.moveTask({ taskId: task.id, listId: targetListId, position: newPosition });
        } else {
          // Move to different list
          this.taskStore.moveTask({ taskId: task.id, listId: targetListId, position: newPosition });
        }
      }
      
      calculatePosition(index: number, listId: string): number {
        const tasks = this.getTasksForList(listId);
        if (tasks.length === 0) return 1;
        if (index === 0) return tasks[0].position / 2;
        if (index >= tasks.length) return tasks[tasks.length - 1].position + 1;
        return (tasks[index - 1].position + tasks[index].position) / 2;
      }
    }
    ```
