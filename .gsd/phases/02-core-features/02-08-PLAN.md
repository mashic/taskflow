    3. Create WebSocketService:
    
    ```typescript
    import { Injectable, inject, signal, OnDestroy } from '@angular/core';
    import { io, Socket } from 'socket.io-client';
    import { Task, List } from '@taskflow/shared-types';
    import { TaskStore } from '../../features/kanban/task.store';
    import { ListStore } from '../../features/kanban/list.store';
    
    @Injectable({ providedIn: 'root' })
    export class WebSocketService implements OnDestroy {
      private socket: Socket | null = null;
      private taskStore = inject(TaskStore);
      private listStore = inject(ListStore);
      
      isConnected = signal(false);
      currentBoardId = signal<string | null>(null);
      
      connect(): void {
        if (this.socket?.connected) return;
        
        this.socket = io('http://localhost:3000/events', {
          transports: ['websocket'],
          autoConnect: true,
        });
        
        this.socket.on('connect', () => {
          this.isConnected.set(true);
          console.log('WebSocket connected');
          // Rejoin room if was viewing a board
          if (this.currentBoardId()) {
            this.joinBoard(this.currentBoardId()!);
          }
        });
        
        this.socket.on('disconnect', () => {
          this.isConnected.set(false);
          console.log('WebSocket disconnected');
        });
        
        // Task events
        this.socket.on('taskCreated', (task: Task) => {
          this.taskStore.addTask(task);
        });
        
        this.socket.on('taskUpdated', (task: Task) => {
          this.taskStore.updateTask(task.id, task);
        });
        
        this.socket.on('taskMoved', (task: Task) => {
          this.taskStore.updateTask(task.id, task);
        });
        
        this.socket.on('taskDeleted', ({ id }: { id: string }) => {
          this.taskStore.removeTask(id);
        });
        
        // List events
        this.socket.on('listCreated', (list: List) => {
          this.listStore.addList(list);
        });
        
        this.socket.on('listUpdated', (list: List) => {
          this.listStore.updateList(list.id, list);
        });
        
        this.socket.on('listDeleted', ({ id }: { id: string }) => {
          this.listStore.removeList(id);
        });
      }
      
      disconnect(): void {
        this.socket?.disconnect();
        this.socket = null;
        this.isConnected.set(false);
      }
      
      joinBoard(boardId: string): void {
        this.currentBoardId.set(boardId);
        this.socket?.emit('joinBoard', boardId);
      }
      
      leaveBoard(boardId: string): void {
        this.socket?.emit('leaveBoard', boardId);
        this.currentBoardId.set(null);
      }
      
      ngOnDestroy(): void {
        this.disconnect();
      }
    }
    ```

    Update KanbanPage to manage WebSocket room lifecycle:
    
    ```typescript
    export class KanbanPage implements OnInit, OnDestroy {
      private wsService = inject(WebSocketService);
      private destroyRef = inject(DestroyRef);
      
      ngOnInit() {
        const id = this.route.snapshot.paramMap.get('id')!;
        this.boardId.set(id);
        
        // Load data
        this.boardStore.setSelectedBoard(id);
        this.listStore.loadListsForBoard(id);
        this.taskStore.loadTasksForBoard(id);
        
        // Connect WebSocket and join room
        this.wsService.connect();
        this.wsService.joinBoard(id);
        
        // Handle route changes (for navigation between boards)
        this.route.paramMap.pipe(
          takeUntilDestroyed(this.destroyRef),
        ).subscribe((params) => {
          const newId = params.get('id');
          if (newId && newId !== this.boardId()) {
            // Leave old room, join new
            this.wsService.leaveBoard(this.boardId()!);
            this.boardId.set(newId);
            this.wsService.joinBoard(newId);
            this.listStore.loadListsForBoard(newId);
            this.taskStore.loadTasksForBoard(newId);
          }
        });
      }
      
      ngOnDestroy() {
        // Leave room when navigating away
        if (this.boardId()) {
          this.wsService.leaveBoard(this.boardId()!);
        }
      }
    }
    ```
    
    Key behaviors:
    - Connect on init, join board room
    - Leave room on destroy
    - Handle board navigation (leave old, join new)
