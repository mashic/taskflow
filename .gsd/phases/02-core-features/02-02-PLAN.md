  <files>apps/web/src/app/features/boards/board.store.ts</files>
  <action>
    Create BoardStore using withEntities pattern from research:
    
    ```typescript
    import { signalStore, withComputed, withMethods, withState } from '@ngrx/signals';
    import { withEntities, addEntity, updateEntity, removeEntity, setAllEntities } from '@ngrx/signals/entities';
    import { Board } from '@taskflow/shared-types';
    
    interface BoardStoreState {
      isLoading: boolean;
      error: string | null;
      selectedBoardId: string | null;
    }
    
    export const BoardStore = signalStore(
      { providedIn: 'root' },
      withEntities<Board>(),
      withState<BoardStoreState>({
        isLoading: false,
        error: null,
        selectedBoardId: null,
      }),
      withComputed(({ entities, selectedBoardId }) => ({
        selectedBoard: computed(() => 
          entities().find(b => b.id === selectedBoardId())
        ),
        boardCount: computed(() => entities().length),
      })),
      withMethods((store, boardService = inject(BoardService)) => ({
        // Sync methods for local state updates
        addBoard(board: Board): void
        updateBoard(id: string, changes: Partial<Board>): void
        removeBoard(id: string): void
        setBoards(boards: Board[]): void
        setSelectedBoard(id: string | null): void
        setLoading(isLoading: boolean): void
        setError(error: string | null): void
        
        // Async methods using rxMethod for API calls
        loadBoards: rxMethod<void>()
        createBoard: rxMethod<CreateBoardDto>()
        // etc.
      }))
    );
    ```
    
    Use rxMethod with tapResponse for API calls (from @ngrx/operators).
    Handle loading states during API operations.
  </action>
