<task type="auto">
  <name>02-05-T3: Create TaskStore with optimistic move</name>
  <files>apps/web/src/app/features/kanban/task.store.ts</files>
  <action>
    Create TaskStore with optimistic update pattern from research:
    
    ```typescript
    interface TaskStoreState {
      currentBoardId: string | null;
      isLoading: boolean;
      error: string | null;
      pendingMoves: Map<string, Task>; // taskId -> original state for rollback
    }
    
    export const TaskStore = signalStore(
      { providedIn: 'root' },
      withEntities<Task>(),
      withState<TaskStoreState>({
        currentBoardId: null,
        isLoading: false,
        error: null,
        pendingMoves: new Map(),
      }),
      withComputed(({ entities }) => ({
        // Get tasks for a specific list, sorted by position
        tasksByList: computed(() => {
          const tasks = entities();
          const grouped = new Map<string, Task[]>();
          for (const task of tasks) {
            const listTasks = grouped.get(task.listId) || [];
            listTasks.push(task);
            grouped.set(task.listId, listTasks);
          }
          // Sort each list's tasks by position
          for (const [listId, listTasks] of grouped) {
            grouped.set(listId, listTasks.toSorted((a, b) => a.position - b.position));
          }
          return grouped;
        }),
      })),
      withMethods((store, taskService = inject(TaskService)) => ({
        loadTasksForBoard: rxMethod<string>(...),
        
        // Optimistic move
        moveTask: rxMethod<{ taskId: string; listId: string; position: number }>(
          pipe(
            tap(({ taskId, listId, position }) => {
              // Store original for rollback
              const original = store.entityMap()[taskId];
              store.pendingMoves().set(taskId, original);
              // Optimistically update
              patchState(store, updateEntity({ id: taskId, changes: { listId, position } }));
            }),
            switchMap(({ taskId, listId, position }) =>
              taskService.moveTask(taskId, { listId, position }).pipe(
                tapResponse({
                  next: (updated) => {
                    patchState(store, updateEntity({ id: taskId, changes: updated }));
                    store.pendingMoves().delete(taskId);
                  },
                  error: () => {
                    // Rollback
                    const original = store.pendingMoves().get(taskId);
                    if (original) {
                      patchState(store, setEntity(original));
                    }
                    store.pendingMoves().delete(taskId);
                  },
                })
              )
            )
          )
        ),
        
        // Local state methods
        addTask(task: Task): void,
        updateTask(id, changes): void,
        removeTask(id): void,
      }))
    );
    ```
    
    Key: pendingMoves Map tracks original state for rollback on error.
