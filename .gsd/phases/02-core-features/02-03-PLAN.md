  <action>
    Create standalone BoardListPage component:
    
    ```typescript
    @Component({
      selector: 'app-board-list',
      standalone: true,
      imports: [RouterLink, ...],
      template: `
        <div class="boards-container">
          <header>
            <h1>My Boards</h1>
            <button (click)="openCreateDialog()">+ New Board</button>
          </header>
          
          @if (boardStore.isLoading()) {
            <p>Loading boards...</p>
          } @else if (boardStore.error()) {
            <p class="error">{{ boardStore.error() }}</p>
          } @else {
            <div class="board-grid">
              @for (board of boardStore.entities(); track board.id) {
                <a [routerLink]="['/boards', board.id]" class="board-card">
                  <h3>{{ board.title }}</h3>
                  <p>{{ board.description }}</p>
                </a>
              } @empty {
                <p>No boards yet. Create your first board!</p>
              }
            </div>
          }
        </div>
      `,
    })
    export class BoardListPage implements OnInit {
      boardStore = inject(BoardStore);
      
      ngOnInit() {
        this.boardStore.loadBoards();
      }
      
      openCreateDialog() { ... }
    }
    ```
    
    Use Angular 21 control flow (@if, @for, @empty).
    Style with CSS grid for board cards.
  </action>
  <verify>Component compiles and renders</verify>
  <done>BoardListPage displays user's boards</done>
