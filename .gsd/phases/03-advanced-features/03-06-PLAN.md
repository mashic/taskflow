<objective>
Create BoardFacade to unify access to all board-related SignalStores using the Facade pattern.

Purpose: Simplify component code by providing single interface to complex store subsystem
Output: BoardFacade service and refactored Kanban page

**Design Pattern: Facade Pattern (GoF)**
- **Definition:** Provide a unified interface to a set of interfaces in a subsystem. Facade defines a higher-level interface that makes the subsystem easier to use.
- **Implementation:** BoardFacade injects BoardStore, ListStore, TaskStore, CommentStore, and ActivityStore. It exposes computed signals that combine data from multiple stores (e.g., isLoading from all stores) and methods that coordinate operations across stores (e.g., loadBoard loads lists, tasks, and activity).
- **Interview talking point:** "I used the Facade pattern to simplify component code. Instead of injecting 5 stores and coordinating their state, components inject one BoardFacade that provides a unified API. This reduces coupling and makes the component code cleaner."
</objective>

    ```typescript
    @Injectable({ providedIn: 'root' })
    export class BoardFacade {
      private readonly boardStore = inject(BoardStore);
      private readonly listStore = inject(ListStore);
      private readonly taskStore = inject(TaskStore);
      private readonly commentStore = inject(CommentStore);
      private readonly activityStore = inject(ActivityStore);
      private readonly teamStore = inject(TeamStore);
      
      // === Unified computed signals ===
      
      // Current board
      readonly currentBoard = this.boardStore.selectedBoard;
      readonly boardId = computed(() => this.currentBoard()?.id ?? null);
      readonly boardTitle = computed(() => this.currentBoard()?.title ?? '');
      
      // Lists and tasks
      readonly lists = this.listStore.sortedLists;
      readonly tasksByList = this.taskStore.tasksByList;
      
      // Loading state (OR across all stores)
      readonly isLoading = computed(() => 
        this.boardStore.isLoading() || 
        this.listStore.isLoading() || 
        this.taskStore.isLoading()
      );
      
      // Error state (first error from any store)
      readonly error = computed(() =>
        this.boardStore.error() ||
        this.listStore.error() ||
        this.taskStore.error()
      );
      
      // Activity
      readonly recentActivity = this.activityStore.recentActivities;
      
      // Team
      readonly teamMembers = this.teamStore.members;
      
      // Combined board data
      readonly boardData = computed(() => ({
        board: this.currentBoard(),
        lists: this.lists(),
        tasksByList: this.tasksByList(),
        isLoading: this.isLoading(),
        error: this.error(),
      }));
      
      // === Board lifecycle ===
      
      loadBoard(boardId: string): void {
        this.boardStore.setSelectedBoard(boardId);
        this.listStore.loadListsForBoard(boardId);
        this.taskStore.loadTasksForBoard(boardId);
        this.activityStore.loadActivities(boardId);
        this.teamStore.loadMembers(boardId);
      }
      
      unloadBoard(): void {
        this.boardStore.setSelectedBoard(null);
        this.listStore.clearLists();
        this.taskStore.clearTasks();
        this.activityStore.clearActivities();
        this.commentStore.clearComments();
      }
      
      // === List operations ===
      
      createList(title: string): void {
        const boardId = this.boardId();
        if (boardId) {
          this.listStore.createList({ title, boardId });
        }
      }
      
      updateList(listId: string, title: string): void {
        this.listStore.updateList(listId, { title });
      }
      
      deleteList(listId: string): void {
        this.listStore.deleteList(listId);
      }
      
      reorderList(listId: string, newPosition: number): void {
        this.listStore.reorderList(listId, newPosition);
      }
      
      // === Task operations ===
      
      createTask(listId: string, title: string, description?: string): void {
        const boardId = this.boardId();
        if (boardId) {
          this.taskStore.createTask({ title, description, listId, boardId });
        }
      }
      
      updateTask(taskId: string, data: Partial<Task>): void {
        this.taskStore.updateTask(taskId, data);
      }
      
      deleteTask(taskId: string): void {
        this.taskStore.deleteTask(taskId);
      }
      
      moveTask(taskId: string, targetListId: string, newPosition: number): void {
        this.taskStore.moveTask({ taskId, targetListId, position: newPosition });
      }
      
      // === Comment operations ===
      
      loadTaskComments(taskId: string): void {
        this.commentStore.loadComments(taskId);
      }
      
      addComment(taskId: string, content: string): void {
        this.commentStore.addComment(taskId, content);
      }
      
      getTaskComments = computed(() => this.commentStore.taskComments());
    }
    ```
