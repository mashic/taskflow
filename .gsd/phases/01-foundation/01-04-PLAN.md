First, check if Angular app already exists. If apps/web/angular.json exists, skip scaffolding.

If needed, scaffold new Angular app:
```bash
cd apps
# Remove existing placeholder if any
rm -rf web
# Generate new Angular 21 app
npx @angular/cli@latest new web --directory=web --style=scss --routing=true --ssr=false --skip-git --package-manager=pnpm
cd web
```

The Angular CLI will create the complete app structure with:
- Standalone components by default
- Vitest for testing
- SCSS for styles
- Routing module

Update apps/web/package.json:
- Change name to "@taskflow/web"
- Add dependency: "@taskflow/shared-types": "workspace:*"

Install dependencies:
```bash
cd apps/web
pnpm install
```

Ensure apps/web/src/app/app.config.ts has proper providers:
```typescript
import { ApplicationConfig, provideZoneChangeDetection } from '@angular/core';
import { provideRouter } from '@angular/router';
import { provideHttpClient, withInterceptors } from '@angular/common/http';
import { routes } from './app.routes';

export const appConfig: ApplicationConfig = {
  providers: [
    provideZoneChangeDetection({ eventCoalescing: true }),
    provideRouter(routes),
    provideHttpClient(),
  ],
};
```

Update apps/web/src/app/app.routes.ts:
```typescript
import { Routes } from '@angular/router';

export const routes: Routes = [
  {
    path: '',
    redirectTo: 'dashboard',
    pathMatch: 'full',
  },
  {
    path: 'login',
    loadComponent: () => import('./features/auth/login.page').then(m => m.LoginPage),
  },
  {
    path: 'register',
    loadComponent: () => import('./features/auth/register.page').then(m => m.RegisterPage),
  },
  {
    path: 'dashboard',
    loadComponent: () => import('./features/dashboard/dashboard.page').then(m => m.DashboardPage),
  },
];
```

Update apps/web/src/app/app.ts (rename from app.component.ts if needed):
```typescript
import { Component } from '@angular/core';
import { RouterOutlet } from '@angular/router';

@Component({
  selector: 'app-root',
  standalone: true,
  imports: [RouterOutlet],
  template: `<router-outlet></router-outlet>`,
})
export class AppComponent {}
```

Ensure apps/web/src/main.ts uses bootstrapApplication:
